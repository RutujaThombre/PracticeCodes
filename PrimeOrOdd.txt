Given number is prime or odd num=7

using System;
namespace practice_code4
{
class program
{
  int number = 7;
  int count = 0;
for (int i=1;i<=number;i++)
{
if (number%i==0)
{
count++;
}
}
if (count==2)
{

Console.WriteLine("prime number");
}
else{
Console.WriteLine("not a prime number");
}
Console.ReadLine();
}
}
}
