swapping

using System;
namespace practice_code
{
class program
{
static void Main(string[]args)
{
int a=10;
int b=20;
int temp;
Console.WriteLine("before swapping the value of a={0},b={1}",a,b);
temp=a;
a=b;
b=temp;
Console.WriteLine("after swapping the value of a={0},b={1}",a,b);
Console.ReadLine();
}
}
}
